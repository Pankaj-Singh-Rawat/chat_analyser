class Node:
    def __init__(self, data):
        self.data, self.next = data, None

class SinglyLinkedList:
    def __init__(self): self.head = None
    def insert(self, data):
        new = Node(data)
        if not self.head: self.head = new; return
        temp = self.head
        while temp.next: temp = temp.next
        temp.next = new
    def delete(self, key):
        temp, prev = self.head, None
        while temp and temp.data != key: prev, temp = temp, temp.next
        if not temp: return
        if not prev: self.head = temp.next
        else: prev.next = temp.next
    def traverse(self):
        temp = self.head
        while temp: print(temp.data, end=" -> "); temp = temp.next
        print("None")

# ===== Doubly Linked List =====
class DNode:
    def __init__(self, data):
        self.data, self.prev, self.next = data, None, None

class DoublyLinkedList:
    def __init__(self): self.head = None
    def insert(self, data):
        new = DNode(data)
        if not self.head: self.head = new; return
        temp = self.head
        while temp.next: temp = temp.next
        temp.next, new.prev = new, temp
    def delete(self, key):
        temp = self.head
        while temp and temp.data != key: temp = temp.next
        if not temp: return
        if temp.prev: temp.prev.next = temp.next
        else: self.head = temp.next
        if temp.next: temp.next.prev = temp.prev
    def traverse(self):
        temp = self.head
        while temp: print(temp.data, end=" <-> "); temp = temp.next
        print("None")

# ===== Circular Linked List =====
class CNode:
    def __init__(self, data):
        self.data, self.next = data, None

class CircularLinkedList:
    def __init__(self): self.head = None
    def insert(self, data):
        new = CNode(data)
        if not self.head:
            self.head = new; new.next = new; return
        temp = self.head
        while temp.next != self.head: temp = temp.next
        temp.next, new.next = new, self.head
    def delete(self, key):
        if not self.head: return
        curr, prev = self.head, None
        while True:
            if curr.data == key:
                if prev: prev.next = curr.next
                else:  # deleting head
                    temp = self.head
                    while temp.next != self.head: temp = temp.next
                    self.head = curr.next; temp.next = self.head
                if curr == curr.next: self.head = None
                return
            prev, curr = curr, curr.next
            if curr == self.head: break
    def traverse(self):
        if not self.head: print("Empty"); return
        temp = self.head
        while True:
            print(temp.data, end=" -> ")
            temp = temp.next
            if temp == self.head: break
        print("(Back to head)")

class SinglyLinkedList:

    def insert(self, data):
        new = Node(data)

        temp = self.head
        while temp.next: temp = temp.next
        temp.next = new

        if not temp: return
        if not prev: self.head = temp.next

# ===== Doubly Linked List =====
class DNode:
    def __init__(self, data):
        self.data, self.prev, self.next = data, None, None

class DoublyLinkedList:
    def __init__(self): self.head = None
    def insert(self, data):
        new = DNode(data)
        if not self.head: self.head = new; return
        temp = self.head
        while temp.next: temp = temp.next
        temp.next, new.prev = new, temp
    def delete(self, key):
        temp = self.head
        while temp and temp.data != key: temp = temp.next
        if not temp: return
        if temp.prev: temp.prev.next = temp.next
        else: self.head = temp.next
        if temp.next: temp.next.prev = temp.prev
    def traverse(self):
        temp = self.head
        while temp: print(temp.data, end=" <-> "); temp = temp.next
        print("None")

# ===== Circular Linked List =====
class CNode:
    def __init__(self, data):
        self.data, self.next = data, None

class CircularLinkedList:
    def __init__(self): self.head = None
    def insert(self, data):
        new = CNode(data)
        if not self.head:
            self.head = new; new.next = new; return
        temp = self.head
        while temp.next != self.head: temp = temp.next
        temp.next, new.next = new, self.head
    def delete(self, key):
        if not self.head: return
        curr, prev = self.head, None
        while True:
            if curr.data == key:
                if prev: prev.next = curr.next
                else:  # deleting head
                    temp = self.head
                    while temp.next != self.head: temp = temp.next
                    self.head = curr.next; temp.next = self.head
                if curr == curr.next: self.head = None
                return
            prev, curr = curr, curr.next
            if curr == self.head: break
    def traverse(self):
        if not self.head: print("Empty"); return
        temp = self.head
        while True:
            print(temp.data, end=" -> ")
            temp = temp.next
            if temp == self.head: break
        print("(Back to head)")

def partition(arr, low, high):
    pivot = arr[high]  # pivot = last element
    i = low - 1

    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]

    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1

def merge(left, right):
    result = []
    i = j = 0
    
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1
    
    # Remaining elements
    result.extend(left[i:])
    result.extend(right[j:])
    
    return result

module full_adder (
    input  a,
    input  b,
    input  cin,
    output sum,
    output cout
);

    assign sum  = a ^ b ^ cin; 
    assign cout = (a & b) | (b & cin) | (a & cin);

endmodule

// 4-BIT RIPPLE CARRY ADDER USING FULL ADDERS
//======================================================
module four_bit_adder (
    input  [3:0] A,
    input  [3:0] B,
    input        Cin,
    output [3:0] Sum,
    output       Cout
);

    wire c1, c2, c3;  // internal carry wires

// Fast power function
public static long power(long base, long exp, long mod) {
    long res = 1;
    while (exp > 0) {
        if ((exp & 1) == 1) res = (res * base) % mod;
        base = (base * base) % mod;
        exp >>= 1;
    }
    return res;
}
